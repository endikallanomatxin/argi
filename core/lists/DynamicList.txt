DynamicList#(.t: Type) : Abstract = (
    append(.self: $&Self, .value: t) -> ()
    insert(.self: $&Self, .index: UIntNative, .value: t) -> ()
    remove(.self: $&Self, .index: UIntNative) -> ()
)

DynamicList#(.t: Type) implements List#(.t: t)

DynamicArray#(.t: Type) implements DynamicList#(.t: t)
SegmentedDynamicArray#(.t: Type) implements DynamicList#(.t: t)
SinglyLinkedList#(.t: Type) implements DynamicList#(.t: t)
DoublyLinkedList#(.t: Type) implements DynamicList#(.t: t)

DynamicArray#(.t: Type) : Type = (
    ---
    A heap allocated dynamic array
    that copies data when length exceeds capacity
    Fast accessing
    ---
    ._data      : &Allocation
    ._data_type : Type        = t
    ._alignment : Alignment   = ..Default
    ._length    : UIntNative
    ._capacity  : UIntNative
)

SegmentedDynamicArray#(.t: Type) : Type = (
    ---
    A heap allocated dynamic array
    that grows by getting more segments of memory, without copying data.
    Slower access than DynamicArray
    ---
    ._allocator : Allocator
    ._data      : &Allocation
    ._data_type : Type        = t
    ._alignment : Alignment   = ..Default
    ._length    : UIntNative
    ._capacity  : UIntNative
)
