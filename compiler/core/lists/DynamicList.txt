DynamicList<t> : Abstract = [
    append(_, _.type)
    insert(_, _.type, Index)
    remove(_, Index)
]

DynamicList<t> implements List<t>

DynamicArray<t> implements DynamicList<t>
SegmentedDynamicArray<t> implements DynamicList<t>
SinglyLinkedList<t> implements DynamicList<t>
DoublyLinkedList<t> implements DynamicList<t>

DynamicArray<t> : Type = [
    ---
    A heap allocated dynamic array
    that copies data when length exceeds capacity
    Fast accessing
    ---
    ._data      : &Allocation
    ._data_type : Type        = t
    ._alignment : Alignment   = ..Default
    ._length    : Int64
    ._capacity  : Int64
]

SegmentedDynamicArray<t> : Type = [
    ---
    A heap allocated dynamic array
    that grows by getting more segments of memory, without copying data.
    Slower access than DynamicArray
    ---
    ._allocator : Allocator
    ._data      : &Allocation
    ._data_type : Type        = t
    ._alignment : Alignment   = ..Default
    ._length    : Int64
    ._capacity  : Int64
]
