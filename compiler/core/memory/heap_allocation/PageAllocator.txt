PageAllocator : Type = (
    ._page_size : UIntNative = os.page_size()
)

alloc(.self: $&PageAllocator, .size: UIntNative, .alignment: Alignment) -> (.allocation: Allocation) !HeapAllocationError := {
    aligned_size ::= align_forward(size, max(self&._page_size, alignment.bytes))
    allocation = os.mmap(aligned_size)
}

resize(.self: $&PageAllocator, .allocation: $&Allocation, .new_size: UIntNative) -> (.ok: Bool) := {
    ok = false  -- no soportado
}

dealloc(.self: $&PageAllocator, .allocation: Allocation) -> () !HeapDeallocationError := {
    os.munmap(allocation.data)
}

PageAllocator implements Allocator
Allocator defaultsto PageAllocator
